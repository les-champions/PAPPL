152 class r_many_t {
153  R r_;
154  S separator_;
155  const size_t min_occurrence_;
156  const size_t max_occurrence_;
157 
158  public:
159  r_many_t(R&& r, S&& separator, size_t min_occurrence, size_t max_occurrence)
160  : r_(std::forward<R>(r)), separator_(std::forward<S>(separator)),
161  min_occurrence_(min_occurrence), max_occurrence_(max_occurrence) {
162  }
163 
164  template<class Iterator>
165  result<Iterator> operator() (Iterator i1, Iterator i2) {
166  auto i_match = r_(i1, i2);
167 
168  if(!i_match.matched)
169  return make_result(!min_occurrence_, i1, i_match.position);
170 
171  size_t count = 1;
172  auto match = i_match;
173 
174  while(match.matched && count < max_occurrence_) {
175  match = separator_(match.position, i2);
176  if(match.matched)
177  match = r_(match.position, i2);
178  if(match.matched) {
179  i_match = match;
180  ++count;
181  }
182  }
183 
184  return make_result(count >= min_occurrence_, i_match.position, match.position);
185  }
186 };
