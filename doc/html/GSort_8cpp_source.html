332 bool GSort::isOver(GSort* otherGSort) {
333  QPointF centerSortMoving(getCenterPoint());
334  QPointF centerSortAround(otherGSort->getCenterPoint());
335  qreal distanceWidthMin = getSizeRect()->width()/2 + otherGSort->getSizeRect()->width()/2 +defaultDistance;
336  qreal distanceHeightMin = getSizeRect()->height()/2 + otherGSort->getSizeRect()->height()/2 +defaultDistance;
337 
338  bool xCond = abs(centerSortMoving.x()-centerSortAround.x())<distanceWidthMin;
339  bool yCond = abs(centerSortMoving.y()-centerSortAround.y())<distanceHeightMin;
340
