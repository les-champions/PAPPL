32 PHPtr PHIO::parse (string const& input) {
33 
34  using namespace axe;
35  PHPtr res = make_shared<PH>();
36 
37  // error
38  auto error = r_fail([](CCHAR i1, CCHAR i2) {});
39 
40  // comment
41  r_rule<const char*> comment;
42  // does not work with malformed comments like (*aaa(*bb*):
43  comment = r_lit("(*") & *(r_any() - r_lit("*)") - r_lit("(*") | comment) & r_lit("*)");
44 
45  // white space
46  auto space = *(r_any(" \t") | comment);
47  auto endl = r_lit("\n");
48  auto trailing_spaces = space & endl;
49 
50  // infinity
51  auto infinity = r_lit("Inf");
52 
53  // process declaration
54  string sortName;
55  int processes;
56  auto sort_name = (r_alpha() | r_char('_')) & *(r_any("_'") | r_alnum());
57  auto sort_declaration = (r_str("process") & space & (sort_name >> sortName) & space & r_ufixed(processes)) >> e_ref([&](CCHAR i1, CCHAR i2) {
58  SortPtr s = Sort::make(sortName, processes);
59  res->addSort(s);
60  });
61  auto sort_declaration_line = sort_declaration & trailing_spaces;
62 
63  // action declaration
64  string actSort1, actSort2;
65  uint actProc1, actProc2, actProc3;
66  double actRate;
67  int actStoch;
68  bool infiniteActRate = false;
69  auto action_required = (sort_name >> actSort1) & space & r_ufixed(actProc1) & space & r_lit("->") & space & (sort_name >> actSort2) & space & r_ufixed(actProc2) & space & r_ufixed(actProc3);
70  auto action_rate = ( (infinity >> [&](CCHAR i1, CCHAR i2) {
71  infiniteActRate = true;
72  })
73  | (r_double(actRate) >> [&](CCHAR i1, CCHAR i2) {
74  infiniteActRate = false;
75  })
76  );
77  auto action_with_rate = action_required & space & r_lit("@") & space & action_rate;
78  auto action_with_stoch = (action_with_rate & space & r_lit("~") & space & r_ufixed(actStoch));
79  auto action = action_with_stoch >> e_ref([&](CCHAR i1, CCHAR i2) {
80  ActionPtr action = make_shared<Action>( res->getSort(actSort1)->getProcess(actProc1)
81  , res->getSort(actSort2)->getProcess(actProc2)
82  , res->getSort(actSort2)->getProcess(actProc3)
83  , infiniteActRate, actRate, actStoch);
84  res->addAction(action);
85  })
86  | action_with_rate >> e_ref([&](CCHAR i1, CCHAR i2) {
87  ActionPtr action = make_shared<Action>( res->getSort(actSort1)->getProcess(actProc1)
88  , res->getSort(actSort2)->getProcess(actProc2)
89  , res->getSort(actSort2)->getProcess(actProc3)
90  , infiniteActRate, actRate
91  , res->getStochasticityAbsorption());
92  res->addAction(action);
93  })
94  | action_required >> e_ref([&](CCHAR i1, CCHAR i2) {
95  ActionPtr action = make_shared<Action>( res->getSort(actSort1)->getProcess(actProc1)
96  , res->getSort(actSort2)->getProcess(actProc2)
97  , res->getSort(actSort2)->getProcess(actProc3)
98  , res->getInfiniteDefaultRate(), res->getDefaultRate()
99  , res->getStochasticityAbsorption());
100  res->addAction(action);
101  });
102  auto action_line = action & trailing_spaces;
103 
104  // body
105  auto body_line = sort_declaration_line | action_line | trailing_spaces ;
106 
107  // footer
108  vector<string> initSorts;
109  vector<int> initProc;
110  auto initial_state = (
111  r_lit("initial_state") & space
112  & r_many((sort_name >> e_push_back(initSorts)) & space & (r_ufixed() >> e_push_back(initProc)), space & r_lit(",") & space)
113  & trailing_spaces
114  ) >> e_ref([&](CCHAR i1, CCHAR i2) {
115  for (unsigned int i=0; i < initSorts.size(); i++)
116  res->getSort(initSorts[i])->setActiveProcess(initProc[i]);
117  initSorts.clear();
118  initProc.clear();
119  });
120  auto footer_line = initial_state | trailing_spaces;
121 
122  // complete file
123  auto body = *body_line;
124  auto footer = *footer_line;
125  auto ph = (body & footer & r_end()) | error;
126  auto result = ph(input.c_str(), input.c_str() + input.length());
127 
128  if (!result.matched)
129  throw ph_parse_error();
130 
131  return res;
132 }

137

139  // (this command transforms complex PH instructions in basic ones)
140  QString phc = "phc";

142  args << "-l" << "dump" << "-i" << QString::fromUtf8(path.c_str()) << "--no-debug";
143  QProcess *phcProcess = new QProcess();
144  phcProcess->start(phc, args);
145  if (!phcProcess->waitForStarted())
146  throw pint_program_not_found() << file_info("phc");
147 
148  // read result
149  QByteArray phcStandardError;
150  QByteArray phcStandardOutput;
151  // Consider phc to have timed out after 10 mins
152  bool timedOut = !phcProcess->waitForFinished(10*60*1000);
153  phcStandardError += phcProcess->readAllStandardError();
154  phcStandardOutput += phcProcess->readAllStandardOutput();
155  delete phcProcess;
156 
157  if (timedOut)
158  throw pint_phc_crash() << (parse_info)"Time out";
159 
160  // parse dump
161  if (!phcStandardError.isEmpty())
162  throw pint_phc_crash() << (parse_info)QString(phcStandardError).toStdString();
163  return parse(QString(phcStandardOutput).toStdString());
164

527 pair<int,int>PHIO::findOrigin( list <pair <int, int> > txy) {

536  int Vy;
537 
538 
539 
540  for(pair<int,int> &i : txy) {
541  moyx=moyx+i.first;
542  S2 = S2+i.first*i.first;
543  moyy=moyy+i.second;
544  S1=S1+i.second*i.second;
545  }
546 
547  moyx=(int)moyx/txy.size();
548  moyy=(int)moyy/txy.size();
549 
550  Vx = S2/txy.size()-moyx*moyx;

552 
553

556 
557  int dmin=fabs(min.first-moyx);
558  origin=std::make_pair(min.first,min.second);
559 
560  for(pair<int,int> &i : txy ) {
561  if (dmin<fabs(i.first-moyx)) {
562  origin= std::make_pair(i.first,i.second);
563  dmin=fabs(i.first-moyx);
564  }
